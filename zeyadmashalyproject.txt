#include <iostream>
#include <string>
using namespace std;

struct Patient {
    int id;
    string name;
    int age;
    string disease;
};

struct Node {
    Patient data;
    Node* next;
};

Node* head = nullptr;
Node* tail = nullptr;

void addPatient(Patient p) {
    Node* newNode = new Node;
    newNode->data = p;
    newNode->next = nullptr;

    if (head == nullptr) {
        head = tail = newNode;
    }
    else {
        tail->next = newNode;
        tail = newNode;
    }
}

void insertAtPosition(Patient p, int pos) {
    Node* newNode = new Node;
    newNode->data = p;
    newNode->next = nullptr;

    if (pos == 0) {
        newNode->next = head;
        head = newNode;
        return;
    }

    Node* temp = head;
    for (int i = 0; i < pos - 1 && temp != nullptr; i++) {
        temp = temp->next;
    }

    if (temp == nullptr) {
        cout << "Position out of range\n";
        delete newNode;
        return;
    }

    newNode->next = temp->next;
    temp->next = newNode;
    if (newNode->next == nullptr)
        tail = newNode;
}

void display() {
    if (head == nullptr) {
        cout << "No patients.\n";
        return;
    }

    Node* temp = head;
    while (temp != nullptr) {
        cout << "ID: " << temp->data.id << endl;
        cout << "Name: " << temp->data.name << endl;
        cout << "Age: " << temp->data.age << endl;
        cout << "Disease: " << temp->data.disease << endl;
        cout << "-------------------\n";
        temp = temp->next;
    }
}

void search(int id) {
    Node* temp = head;

    while (temp != nullptr) {
        if (temp->data.id == id) {
            cout << "Found: " << temp->data.name << endl;
            return;
        }
        temp = temp->next;
    }

    cout << "Not found\n";
}

void deletePatient(int id) {
    if (head == nullptr) {
        tail = nullptr;
        cout << "List empty\n";
        return;
    }

    if (head->data.id == id) {
        Node* temp = head;
        head = head->next;

        if (head == nullptr)
            tail = nullptr;
        delete temp;
        cout << "Deleted\n";
        return;
    }

    Node* curr = head;
    Node* prev = nullptr;

    while (curr != nullptr && curr->data.id != id) {
        prev = curr;
        curr = curr->next;
    }

    if (curr == nullptr) {
        cout << "Not found\n";
        return;
    }


    if (curr == tail) {
        tail = prev;
    }

    prev->next = curr->next;
    delete curr;
    cout << "Deleted\n";
}

void sortList() {
    if (head == nullptr) return;

    bool swapped;
    Node* ptr;

    do {
        swapped = false;
        ptr = head;

        while (ptr->next != nullptr) {
            if (ptr->data.id > ptr->next->data.id) {
                Patient temp = ptr->data;
                ptr->data = ptr->next->data;
                ptr->next->data = temp;
                swapped = true;
            }
            ptr = ptr->next;
        }

    } while (swapped);

    cout << "List sorted by ID\n";
}
